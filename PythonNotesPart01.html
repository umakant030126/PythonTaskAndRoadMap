Set Operations math
A = {1, 2, 3, 4, 5}
B = {4, 5, 6, 7, 8}

# Union — all elements from both
A | B  == A.union(B)          # {1,2,3,4,5,6,7,8}

# Intersection — elements in BOTH
A & B  == A.intersection(B)   # {4, 5}

# Difference — in A but NOT in B
A - B  == A.difference(B)     # {1, 2, 3}

# Symmetric difference — in A or B but NOT both
A ^ B  == A.symmetric_difference(B)  # {1,2,3,6,7,8}

# Subset / Superset
{1,2}.issubset(A)      # True  ({1,2} ⊆ A)
A.issuperset({1,2})   # True  (A ⊇ {1,2})
A.isdisjoint({9,10}) # True  (no common elements)

# In-place variants
A |= B  # A.update(B)
A &= B  # A.intersection_update(B)
A -= B  # A.difference_update(B)

# Remove duplicates from a list
lst = [1,2,2,3,3,4]
unique = list(set(lst))  # [1,2,3,4] (order may vary)

# Custom encoder for unsupported types
from datetime import date

class DateEncoder(json.JSONEncoder):
    def default(self, obj):
        if isinstance(obj, date):
            return obj.isoformat()
        return super().default(obj)

payload = {"date": date(2025, 1, 15)}
json.dumps(payload, cls=DateEncoder)
# '{"date": "2025-01-15"}'

import json, os

def load_config(path="config.json"):
    defaults = {"debug": False, "port": 8080}
    if os.path.exists(path):
        with open(path) as f:
            return {**defaults, **json.load(f)}
    return defaults

File Modes Reference modes
ModeDescription
'r'Read (default). Error if file doesn't exist.
'w'Write. Creates file, OVERWRITES if exists.
'a'Append. Creates file, adds to end if exists.
'x'Exclusive create. Error if file already exists.
'r+'Read + Write. File must exist.
'w+'Write + Read. Truncates file.

⚠️ Warning: 'w' mode erases all existing content! Use 'a' to append.

    except (FileNotFoundError, IOError):
